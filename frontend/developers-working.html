import React from 'react';
import { NeroWidget } from '@nero-protocol/react-sdk';
import '@nero-protocol/react-sdk/dist/styles.css';

function App() {
  return (
    <div className="app">
      {/* Your dApp content */}
      <h1>My DeFi Protocol</h1>
      
      {/* Nero AI Widget */}
      <NeroWidget
        platformId="your_platform_id"
        theme="light"
        primaryColor="#6366f1"
        position="bottom-right"
        apiEndpoint="https://api.nero.ai/v1"
        onMessage={(msg) => console.log('User asked:', msg)}
      />
    </div>
  );
}

export default App;

        theme: 'light'
      }
    }
  },

    enabled: true,
    freeQueries: 10, // Per day
    queryPrice: '0.001', // MOVE tokens
    treasury: '0x...' // Your treasury address
  },
  
  // NFT Agent
  nft: {
    enabled: true,
    contract: '0x...', // Nero NFT contract
    autoMint: true
  },

  onWalletConnect: (address) => console.log('Wallet:', address),

  onXPGained: (xp) => console.log('XP gained:', xp),
  onLevelUp: (level) => console.log('Level up!', level)
};

// Initialize
const nero = new NeroWidget(neroConfig);
